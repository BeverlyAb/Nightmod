Sets the position of the camera in screen.

(defn position!
  ([screen vec-3]
    (position! screen (x vec-3) (y vec-3) (z vec-3)))
  ([screen x-val y-val]
    (position! screen x-val y-val nil))
  ([screen x-val y-val z-val]
    (some->> x-val (x! screen))
    (some->> y-val (y! screen))
    (some->> z-val (z! screen))))
