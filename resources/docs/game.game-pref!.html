(game-pref! & keyvals)

Stores preferences.

(game-pref! {:level 10 :health 5}) ; replace all preferences
(game-pref! :level 10 :health 5) ; add to existing preferences

(defn game-pref!
  [& keyvals]
  (->> (if (map? (first keyvals))
         (first keyvals)
         (apply assoc (game-pref) keyvals))
       pr-str
       (spit (io/file @u/project-dir u/prefs-file))))
