(defn image*
  [arg]
  (ActorEntity.
    (cond
      (map? arg)
      (Image. ^TextureRegion (:object arg))
      (string? arg)
      (Image. (or (u/load-asset arg Texture)
                  (Texture. ^String arg)))
      :else
      (Image. arg))))
(defmacro image
  [arg & options]
  `(let [entity# (image* ~arg)]
     (u/calls! ^Image (u/get-obj entity# :object) ~@options)
     entity#))
