(defn map-layers*
  ([]
    (MapLayers.))
  ([screen]
    (let [^BatchTiledMapRenderer renderer (u/get-obj screen :renderer)]
      (-> renderer .getMap .getLayers))))
(defmacro map-layers
  [screen & options]
  `(let [^MapLayers object# (map-layers* ~screen)]
     (u/calls! object# ~@options)))
