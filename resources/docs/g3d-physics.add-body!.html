(defn add-body!
  [screen body]
  (cond
    (instance? btRigidBody (:object body))
    (bullet-3d! screen :add-rigid-body (:object body))
    (instance? btSoftBody (:object body))
    (bullet-3d! screen :add-soft-body (:object body)))
  body)
