(disconnect! client)

Closes the sockets and interrupts the receiver thread. The client is a hash map returned by the client function, or a play-clj screen map that contains a client hash map associated with the :network key.

(let [screen (update! screen :network (client screen))]
  (disconnect! screen))
Source
(defn disconnect!
  [client]
  (doto (get-obj client :network :context)
    (.destroySocket (get-obj client :network :sender))
    (.destroySocket (get-obj client :network :receiver)))
  (.interrupt (get-obj client :network :thread)))
